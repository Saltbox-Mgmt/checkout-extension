class SFCCDebuggerPanel { constructor() { this.requirements = [ { key: "shippingAddress", label: "Shipping Address", required: true }, { key: "deliveryMethod", label: "Delivery Method", required: true }, { key: "inventory", label: "Inventory", required: false }, { key: "taxes", label: "Taxes", required: false }, { key: "billingAddress", label: "Billing Address", required: true }, { key: "payment", label: "Payment", required: true }, ] this.networkCalls = [] this.errors = [] this.debugLogs = [] this.checkoutData = {} this.activeTab = "network" this.sessionStart = Date.now() this.analyzer = new window.CommerceOnCoreAnalyzer() this.init() } init() { this.setupEventListeners() this.startNetworkMonitoring() this.renderRequirements() this.renderContent() this.updateSessionInfo() // Update every second setInterval(() => this.updateSessionInfo(), 1000) } setupEventListeners() { // Tab switching document.querySelectorAll(".tab").forEach((tab) => { tab.addEventListener("click", (e) => { this.switchTab(e.target.dataset.tab) }) }) // Clear button document.getElementById("clear-btn").addEventListener("click", () => { this.clearData() }) // Export button document.getElementById("export-btn").addEventListener("click", () => { this.exportData() }) } startNetworkMonitoring() { // Listen for network events from content script window.chrome.runtime.onMessage.addListener((message, sender, sendResponse) => { if (message.type === "networkCall") { this.handleNetworkCall(message.data) } else if (message.type === "error") { this.handleError(message.data) } else if (message.type === "debugLog") { this.handleDebugLog(message.data) } }) // Request current tab to start monitoring window.chrome.tabs.query({ active: true, currentWindow: true }, (tabs) => { if (tabs[0]) { window.chrome.tabs.sendMessage(tabs[0].id, { action: "startMonitoring" }) } }) } handleNetworkCall(callData) { this.networkCalls.push({ ...callData, timestamp: Date.now(), id: this.generateId(), }) // Analyze call for checkout requirements this.analyzeCheckoutCall(callData) this.renderContent() this.renderRequirements() this.updateSessionInfo() } handleError(errorData) { this.errors.push({ ...errorData, timestamp: Date.now(), id: this.generateId(), }) this.renderContent() this.updateSessionInfo() } handleDebugLog(logData) { this.debugLogs.push({ ...logData, timestamp: Date.now(), id: this.generateId(), }) this.renderContent() } analyzeCheckoutCall(callData) { const url = callData.url.toLowerCase() const response = callData.response // Commerce on Core specific analysis if (url.includes("/shipping-address") || url.includes("/delivery-address")) { if (callData.status < 400 && response && !response.errors) { this.checkoutData.shippingAddress = true } } if (url.includes("/delivery-methods")) { if (callData.status < 400 && response && response.deliveryMethods) { this.checkoutData.deliveryMethod = true } } if (url.includes("/inventory") || url.includes("/cart-items")) { if (callData.status < 400) { this.checkoutData.inventory = true } } if (url.includes("/taxes")) { if (callData.status < 400 && response && response.taxes) { this.checkoutData.taxes = true } } if (url.includes("/billing-address")) { if (callData.status < 400 && response && !response.errors) { this.checkoutData.billingAddress = true } } if (url.includes("/payments")) { if (callData.status < 400 && response && !response.errors) { this.checkoutData.payment = true } else if (response?.errors) { // Payment failed - mark as incomplete and store error details this.checkoutData.payment = false this.checkoutData.paymentError = { errors: response.errors, salesforceResultCode: response.salesforceResultCode, paymentToken: this.extractPaymentToken(callData.requestBody), } } } } extractPaymentToken(payload) { try { const parsedPayload = typeof payload === "string" ? JSON.parse(payload) : payload return parsedPayload?.paymentToken } catch (e) { return null } } switchTab(tabName) { this.activeTab = tabName // Update tab buttons document.querySelectorAll(".tab").forEach((tab) => { tab.classList.toggle("active", tab.dataset.tab === tabName) }) this.renderContent() } renderRequirements() { const container = document.getElementById("requirements-list") container.innerHTML = "" this.requirements.forEach((req) => { const item = document.createElement("div") item.className = "requirement-item" const status = this.getRequirementStatus(req.key, req.required) const statusClass = status === "complete" ? "status-complete" : status === "optional" ? "status-optional" : "status-incomplete" item.innerHTML = `
${req.label} ` container.appendChild(item) }) } getRequirementStatus(key, required) { const hasData = this.checkoutData[key] if (hasData) return "complete" if (!required) return "optional" return "incomplete" } renderContent() { const container = document.getElementById("tab-content") switch (this.activeTab) { case "network": container.innerHTML = this.renderNetworkCalls() break case "errors": container.innerHTML = this.renderErrors() break case "logs": container.innerHTML = this.renderDebugLogs() break } // Add click handlers for expandable items container.querySelectorAll(".network-call-header").forEach((header) => { header.addEventListener("click", (e) => { const call = e.target.closest(".network-call") call.classList.toggle("expanded") }) }) } renderNetworkCalls() { if (this.networkCalls.length === 0) { return `
üåê
No network calls captured yet
Navigate to a Commerce on Core checkout page to start monitoring
` } return this.networkCalls .map( (call) => `
${call.method} ${this.truncateUrl(call.url)} ${call.checkoutStage ? `${call.checkoutStage}` : ""}
${call.webstoreId ? `Store: ${call.webstoreId.substring(0, 8)}...` : ""}
${call.status}
${ call.checkoutId ? `
Checkout Session
Checkout ID: ${call.checkoutId}
` : "" } ${ call.paymentErrors ? `
Payment Errors
${JSON.stringify(call.paymentErrors, null, 2)}
` : "" } ${ call.salesforceResultCode ? `
Salesforce Result Code
${call.salesforceResultCode}
` : "" }
Request URL
${call.url}
${ call.requestBody ? `
Request Payload
${typeof call.requestBody === "string" ? call.requestBody : JSON.stringify(call.requestBody, null, 2)}
` : "" }
Response
${JSON.stringify(call.response || {}, null, 2)}
Timing
Duration: ${call.duration || 0}ms
`, ) .join("") } renderErrors() { if (this.errors.length === 0) { return `
‚úÖ
No errors detected
` } return this.errors .map( (error) => `
${error.message}
${new Date(error.timestamp).toLocaleTimeString()}
Stack Trace
${error.stack || "No stack trace available"}
`, ) .join("") } renderDebugLogs() { if (this.debugLogs.length === 0) { return `
üìù
No debug logs captured
Debug logs will appear here when available
` } return this.debugLogs .map( (log) => `
${log.message}
${new Date(log.timestamp).toLocaleTimeString()}
Details
${JSON.stringify(log.data || {}, null, 2)}
`, ) .join("") } updateSessionInfo() { const duration = Math.floor((Date.now() - this.sessionStart) / 1000) document.getElementById("call-count").textContent = this.networkCalls.length document.getElementById("error-count").textContent = this.errors.length document.getElementById("session-duration").textContent = `${duration}s` } clearData() { this.networkCalls = [] this.errors = [] this.debugLogs = [] this.checkoutData = {} this.sessionStart = Date.now() this.renderContent() this.renderRequirements() this.updateSessionInfo() } exportData() { const data = { // Core monitoring data networkCalls: this.networkCalls, errors: this.errors, debugLogs: this.debugLogs, checkoutData: this.checkoutData, requirements: this.requirements, // Salesforce integration data salesforceLogs: this.salesforceLogs || [], correlations: this.correlations || [], // Session metadata exportTime: new Date().toISOString(), sessionStart: this.sessionStart, sessionDuration: Date.now() - this.sessionStart, // Enhanced summary summary: { totalNetworkCalls: this.networkCalls.length, totalErrors: this.errors.length, totalDebugLogs: this.debugLogs.length, totalSalesforceLogs: (this.salesforceLogs || []).length, totalCorrelations: (this.correlations || []).length, checkoutRequirements: { completed: this.requirements.filter((req) => this.getRequirementStatus(req.key, req.required) === "complete") .length, total: this.requirements.length, }, }, // Analysis data analysis: this.analyzer ? this.analyzer.generateDebugReport(this.networkCalls) : null, // Export metadata metadata: { exportedFrom: "devtools-panel", extensionVersion: "1.0.0", exportFormat: "sfcc-debug-v1", }, } const blob = new Blob([JSON.stringify(data, null, 2)], { type: "application/json" }) const url = URL.createObjectURL(blob) const a = document.createElement("a") a.href = url a.download = `sfcc-debug-${Date.now()}.json` a.click() URL.revokeObjectURL(url) console.log("üìÅ DevTools panel export completed:", { networkCalls: data.networkCalls.length, salesforceLogs: data.salesforceLogs.length, correlations: data.correlations.length, }) } truncateUrl(url) { if (url.length > 60) { return url.substring(0, 57) + "..." } return url } generateId() { return Math.random().toString(36).substring(2, 15) } } // Initialize the panel when DOM is loaded document.addEventListener("DOMContentLoaded", () => { new SFCCDebuggerPanel() })
